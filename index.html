//+------------------------------------------------------------------+
//| Insertion Sort Algorithm |
//+------------------------------------------------------------------+
void InsertionSort(double &array[])
{
int n = ArraySize(array);

for(int i = 1; i < n; i++)
{
double key = array[i];
int j = i - 1;

// Move os elementos do array[0..i-1] que são maiores que key
// para uma posição à frente de sua posição atual
while(j >= 0 && array[j] > key)
{
array[j + 1] = array[j];
j--;
}
array[j + 1] = key;
}
}

//+------------------------------------------------------------------+
//| Exemplo de uso do Insertion Sort |
//+------------------------------------------------------------------+
void OnStart()
{
double prices[] = {1.2345, 1.2321, 1.2356, 1.2300, 1.2389, 1.2400};
int size = ArraySize(prices);

Print("Array antes da ordenação:");

}
